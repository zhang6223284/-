#### 一些有趣的例子

```javascript
var a = {n:1};  
var b = a; // 持有a，以回查  
a.x = a = {n:2};  // 这一步是从右往左执行的
alert(a.x);// --> undefined  
alert(b.x);// --> [object Object] 


function changeStuff(a, b, c)
{
  a = a * 10;
  b.item = "changed";
  c = {item: "changed"};
}

var num = 10;
var obj1 = {item: "unchanged"};
var obj2 = {item: "unchanged"};

changeStuff(num, obj1, obj2);

console.log(num);
console.log(obj1.item);    
console.log(obj2.item);

// 输出结果
10
changed
unchanged
```
